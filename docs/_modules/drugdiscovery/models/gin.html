class GraphIsomorphismNetwork(nn.Module, core.Configurable):
    # TODO: format and clean up
    """
    Graph Ismorphism Network proposed in `How Powerful are Graph Neural Networks?`_

    .. _How Powerful are Graph Neural Networks?:
        https://arxiv.org/pdf/1810.00826.pdf

    Parameters:
        input_dim (int): input dimension
        hidden_dims (list of int): hidden dimensions
        emb_dim (int): embedding dim (this is for the pre-training GIN only)
        num_layer (int): number of layers (this is for the pre-training GIN only)
        edge_input_dim (int, optional): dimension of edge features
        num_mlp_layer (int, optional): number of MLP layers
        eps (int, optional): initial epsilon
        learn_eps (bool, optional): learn epsilon or not
        short_cut (bool, optional): use short cut or not
        batch_norm (bool, optional): apply batch normalization or not
        dropout_ratio (float, optional): apply drop-out
        activation (str or function, optional): activation function
        concat_hidden (bool, optional): concat hidden representations from all layers as output
        GIN_version (str, optional): by default it is the standard GIN model,
        `   and one specialized GIN model is for the pre-training only
    """

    def __init__(self, input_dim=None, hidden_dims=None, emb_dim=None, num_layer=None,
                 edge_input_dim=None, num_mlp_layer=2, eps=0, learn_eps=False,
                 short_cut=False, batch_norm=False, dropout_ratio=0., activation="relu", concat_hidden=False, readout="sum",
                 GIN_version="standard"):
        super(GraphIsomorphismNetwork, self).__init__()

        self.GIN_version = GIN_version
        if self.GIN_version == "standard":
            if not isinstance(hidden_dims, container_abcs.Sequence):
                hidden_dims = [hidden_dims]
            self.input_dim = input_dim
            self.output_dim = hidden_dims[-1] * (len(hidden_dims) if concat_hidden else 1)
            self.dims = [input_dim] + list(hidden_dims)

            self.layers = nn.ModuleList()
            for i in range(len(self.dims) - 1):
                layer_hidden_dims = [self.dims[i + 1]] * (num_mlp_layer - 1)
                self.layers.append(layers.GraphIsomorphismConv(self.dims[i], self.dims[i + 1], edge_input_dim,
                                                               layer_hidden_dims, eps, learn_eps, batch_norm,
                                                               activation))
        elif self.GIN_version == "pretraining":
            num_atom_type = 120  # including the extra mask tokens
            num_chirality_tag = 3

            self.emb_dim = emb_dim
            self.num_layer = num_layer
            self.output_dim = emb_dim * (num_layer if concat_hidden else 1)

            self.x_embedding1 = torch.nn.Embedding(num_atom_type, emb_dim)
            self.x_embedding2 = torch.nn.Embedding(num_chirality_tag, emb_dim)

            torch.nn.init.xavier_uniform_(self.x_embedding1.weight.data)
            torch.nn.init.xavier_uniform_(self.x_embedding2.weight.data)

            self.layers = nn.ModuleList()
            for i in range(self.num_layer):
                self.layers.append(layers.GraphIsomorphismConvPretraining(
                    emb_dim=self.emb_dim, activation=activation, dropout_ratio=dropout_ratio, batch_norm=batch_norm))

            self.batch_norms = torch.nn.ModuleList()
            for layer in range(num_layer):
                self.batch_norms.append(torch.nn.BatchNorm1d(emb_dim))

        else:
            assert ValueError("GIN version %s is not included." % self.GIN_version)

        self.edge_input_dim = edge_input_dim
        self.short_cut = short_cut
        self.concat_hidden = concat_hidden
        self.dropout_ratio = dropout_ratio

        if readout == "sum":
            self.readout = layers.SumReadout()
        elif readout == "mean":
            self.readout = layers.MeanReadout()
        else:
            raise ValueError("Unknown readout `%s`" % readout)

    def forward(self, graph, input, edge_attr=None, all_loss=None, metric=None):
        # TODO: @Shengchao. If edge_attr always refers to graph.edge_feature, let's remove it for consistency.
        # You can access edge_attr from graph.edge_feature.
        # This is how it is implemented in most other GNNs in the platform.
        """"""
        hiddens = []
        if self.GIN_version == "pretraining":
            # TODO: This is dangerous.
            # If a user passes a float tensor (e.g. a wrong input), this will not raise any error.
            input = input.long()
            if edge_attr is None:
                edge_attr = graph.edge_feature
            # TODO: same as above
            edge_attr = edge_attr.long()
            layer_input = self.x_embedding1(input[:, 0]) + self.x_embedding2(input[:, 1])
        else:
            layer_input = input

        for layer_idx, layer in enumerate(self.layers):

            if self.GIN_version == "standard":
                hidden = layer(graph, layer_input)

            elif self.GIN_version == "pretraining":
                hidden = layer(graph, layer_input, edge_attr)
                hidden = self.batch_norms[layer_idx](hidden)
                # last layer doesn't have activation
                # TODO: this makes `hiddens` inconsistent, and affects graph feature
                if layer_idx < len(self.layers)- 1:
                    hidden = F.relu(hidden)
                hidden = F.dropout(hidden, self.dropout_ratio, training=self.training)

            if self.short_cut and hidden.shape == layer_input.shape:
                hidden = hidden + layer_input
            hiddens.append(hidden)
            layer_input = hidden

        if self.concat_hidden:
            node_feature = torch.cat(hiddens, dim=-1)
        else:
            node_feature = hiddens[-1]
        graph_feature = self.readout(graph, node_feature)

        return {
            "graph_feature": graph_feature,
            "node_feature": node_feature
        }
