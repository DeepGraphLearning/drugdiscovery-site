class EdgePrediction(tasks.Task, core.Configurable):
    """
    Edge Prediction as pre-training for graph-like data.

    .. _Inductive Representation Learning on Large Graphs:
        https://arxiv.org/abs/1706.02216

    Parameters:
        model (nn.Module): graph neural network model
    """

    # _option_members = ['criterion', 'metric']

    def __init__(self, model, verbose=0):
        super(EdgePrediction, self).__init__()
        self.model = model
        self.verbose = verbose
        self.criterion = nn.BCEWithLogitsLoss()

    def feed_into_network(self, batch):
        graph = batch['graph']
        x = graph.node_feature.long()
        edge_feature = graph.edge_feature.long()
        output = self.model(graph, x, edge_feature)
        node_feature = output['node_feature']
        return node_feature

    def negative_edge_sampling(self, batch):
        graph = batch['graph']

        num_graph = len(graph.num_nodes)
        neg_edge_per_graph = int(len(graph.edge_list) / (num_graph * 2))
        negative_edge_list = \
            torch.rand(num_graph, neg_edge_per_graph, 2, device=graph.device).double() * graph.num_nodes.unsqueeze(
                1).unsqueeze(2) \
            + (graph.num_cum_nodes - graph.num_nodes).unsqueeze(1).unsqueeze(2)
        negative_edge_list = (negative_edge_list.reshape(-1, 2)).long()

        return negative_edge_list

    def predict_and_target(self, batch):
        node_embed = self.feed_into_network(batch)
        edge_list = batch['graph'].edge_list
        negative_edge_list = self.negative_edge_sampling(batch)

        # TODO: shouldn't assume edges are stepped by 2
        positive_score = torch.sum(node_embed[edge_list[::2, 0]] * node_embed[edge_list[::2, 1]], dim=1)
        negative_score = torch.sum(node_embed[negative_edge_list[:, 0]] * node_embed[negative_edge_list[:, 1]], dim=1)

        pred = torch.cat([positive_score, negative_score], dim=0)
        target = torch.cat([torch.ones_like(positive_score), torch.zeros_like(negative_score)], dim=0)
        return pred, target

    def evaluate(self, pred, target):
        metric = {}
        acc = torch.mean(((pred > 0).float() == (target == 1).float()).float())
        metric['acc'] = acc
        return metric


[docs]    def forward(self, batch):
        metric = {}

        pred, target = self.predict_and_target(batch)
        loss = self.criterion(pred, target)

        name = tasks._get_criterion_name(self.criterion)
        metric[name] = loss
        metric.update(self.evaluate(pred, target))

        return loss, metric

[docs]class Discriminator(nn.Module):
    def __init__(self, hidden_dim):
        super(Discriminator, self).__init__()
        self.weight = nn.Parameter(torch.Tensor(hidden_dim, hidden_dim))
        self.reset_parameters()

    def reset_parameters(self):
        size = self.weight.size(0)
        self.uniform(size, self.weight)


[docs]    def forward(self, x, summary):
        h = torch.matmul(summary, self.weight)
        return torch.sum(x*h, dim=1)


    def uniform(self, size, tensor):
        if tensor is not None:
            bound = 1.0 / math.sqrt(size)
            tensor.data.uniform_(-bound, bound)

class DeepGraphInfoMax(tasks.Task, core.Configurable):
    """
    Deep Graph InfoMax as pre-training for graph-like data.

    .. _Deep Graph Infomax:
        https://arxiv.org/abs/1809.10341

    Parameters:
        model (nn.Module): graph neural network model
        proj_model (nn.Module): projection head model
        discriminator (nn.Module): contrastive model
    """

    # _option_members = ['criterion', 'metric']

    def __init__(self, model, proj_model, discriminator, verbose=0):
        super(DeepGraphInfoMax, self).__init__()
        self.model = model
        self.proj_model = proj_model
        self.discriminator = discriminator
        self.verbose = verbose
        self.criterion = nn.BCEWithLogitsLoss()

    def feed_into_network(self, graph):
        x = graph.node_feature.long()
        edge_feature = graph.edge_feature.long()
        output = self.model(graph, x, edge_feature)
        # TODO: bug: last layer doesn't have activation
        node_feature = self.proj_model(output['node_feature'])
        # TODO: given the above bug, the graph feature could be problematic
        graph_feature = self.proj_model(output['graph_feature'])
        return node_feature, graph_feature

    def cycle_index(self, num, shift):
        arr = torch.arange(num) + shift
        arr[-shift:] = torch.arange(shift)
        return arr

    def predict_and_target(self, batch):
        graph = batch['graph']
        node_emb, graph_emb = self.feed_into_network(graph)

        positive_expanded_summary_emb = graph_emb[graph.node2graph]

        shifted_summary_emb = graph_emb[self.cycle_index(len(graph_emb), 1)]
        negative_expanded_summary_emb = shifted_summary_emb[graph.node2graph]

        positive_score = self.discriminator(node_emb, positive_expanded_summary_emb)
        negative_score = self.discriminator(node_emb, negative_expanded_summary_emb)
        pred = torch.cat([positive_score, negative_score], dim=0)
        target = torch.cat([torch.ones_like(positive_score), torch.zeros_like(negative_score)], dim=0)
        return pred, target

    def evaluate(self, pred, target):
        metric = {}
        acc = torch.mean(((pred > 0).float() == (target == 1).float()).float())
        metric['acc'] = acc
        return metric

class AttributeMasking(tasks.Task, core.Configurable):
    """
    Attribute Masking as pre-training for graph-like data.

    .. _Strategies for Pre-training Graph Neural Networks:
        https://arxiv.org/abs/1905.12265

    Parameters:
        model (nn.Module): graph neural network model
        atom_attribute_model (nn.Module): atom (attribute) prediction model
    """
    _option_members = ['criterion', 'metric']

    def __init__(self, model, atom_attribute_model, mask_rate, verbose=0):
        super(AttributeMasking, self).__init__()
        self.model = model
        self.atom_attribute_model = atom_attribute_model
        self.mask_rate = mask_rate
        self.verbose = verbose
        self.criterion = nn.CrossEntropyLoss()

    def feed_into_network(self, graph):
        x = graph.masked_node_feature.long()
        edge_feature = graph.edge_feature.long()
        output = self.model(graph, x, edge_feature)
        node_feature = output['node_feature']
        return node_feature

    def predict_and_target(self, batch):
        graph = batch['graph']
        node_actual = graph.node_feature[:, 0]
        num_node = graph.num_node
        num_graph = len(graph.num_nodes)
        num_masked_node = int(num_node * self.mask_rate) + num_graph

        masked_node_list = (torch.rand(num_masked_node, device=graph.device) * num_node).long()
        target = node_actual[masked_node_list]

        graph.masked_node_feature = copy.deepcopy(graph.node_feature)
        graph.masked_node_feature[:, 0].index_fill_(0, masked_node_list, 119)
        graph.masked_node_feature[:, 1].index_fill_(0, masked_node_list, 0)

        node_embed = self.feed_into_network(graph)
        node_embed = node_embed[masked_node_list]
        pred = self.atom_attribute_model(node_embed)

        return pred, target

    def evaluate(self, pred, target):
        metric = {}
        metric['acc'] = torch.sum(torch.max(pred.detach(), dim=1)[1] == target) / len(pred)
        return metric

class ContextPrediction(tasks.Task, core.Configurable):
    """
    Context Prediction as pre-training for graph-like data.

    .. _Strategies for Pre-training Graph Neural Networks:
        https://arxiv.org/abs/1905.12265

    Parameters:
        model (nn.Module): graph neural network model
        model_context (nn.Module): graph neural network model for context prediction
        model_context_readout (nn.Module): graph aggregation function for the context prediction model
        k (int) : radius for the K-hop neighborhood
        l1 (int) : radius for the l1-hop neighborhood, context graph is the subgraph between l1- and l2-hop from the center node.
        l2 (int) : radius for the l2-hop neighborhood, context graph is the subgraph between l1- and l2-hop from the center node.
        neg_sample_ratio (int) : the number of negative substructures to each positive substructure
    """
    _option_members = ['criterion', 'metric']

    def __init__(self, model, model_context, model_context_readout, k, l1, l2, neg_sample_ratio, verbose=0):
        super(ContextPrediction, self).__init__()
        self.model = model
        self.model_context = model_context
        self.model_context_readout = model_context_readout
        self.k = k
        self.l1 = l1
        self.l2 = l2
        self.neg_sample_ratio = neg_sample_ratio
        self.verbose = verbose
        self.criterion = nn.BCEWithLogitsLoss()

    def cycle_index(self, num, shift):
        arr = torch.arange(num) + shift
        arr[-shift:] = torch.arange(shift)
        return arr

    def extract_substruct_context_graph(self, batch):
        graph = batch['graph']
        num_node = graph.num_node
        edge_list = graph.edge_list
        node_u = edge_list[:, 0]
        node_v = edge_list[:, 1]

        num_graph = len(graph.num_nodes)
        relative_node_index = (torch.rand(num_graph, 1, device=graph.device) * graph.num_nodes.unsqueeze(-1)).long()[:, 0]
        # ### for debugging only
        # relative_node_index = torch.zeros_like(graph.num_nodes).long()
        center_node_idx = relative_node_index + graph.num_cum_nodes - graph.num_nodes
        dist = torch.full((num_node,), num_node, device=graph.device, dtype=torch.int32)
        dist[center_node_idx] = 0

        for _ in range(self.l2):
            dist_v2u = dist[node_v]
            # This might be a bug is there's isolated node, i.e., molecule with only one atom
            # dist_u = 1 + scatter_min(dist_v2u, node_u, dim=0, dim_size=num_node)[0]
            # dist = dist.min(dist_u)
            dist_u = scatter_min(dist_v2u, node_u, dim=0, dim_size=num_node)[0]
            dist = 1 + (dist - 1).min(dist_u)

        substruct_node_masking = dist <= self.k
        l1_node_masking = dist <= self.l1
        l2_node_masking = dist <= self.l2

        context_node_masking = torch.logical_xor(l1_node_masking, l2_node_masking)
        context_substruct_overlap_masking = (substruct_node_masking & context_node_masking)

        center_node_masking = torch.zeros((graph.num_node), device=graph.device).bool()
        center_node_masking[center_node_idx] = True

        graph.meta_dict = copy.deepcopy(graph.meta_dict)
        with graph.node():
            graph.center_node_masking = center_node_masking
            graph.context_substruct_overlap_masking = context_substruct_overlap_masking

        context_masking_graph = graph.subgraph(context_node_masking)
        masking_graph_index = context_masking_graph.num_nodes > 0
        valid_graph_index = masking_graph_index.nonzero()[:, 0]

        substruct_graph = graph.subgraph(substruct_node_masking).graph_mask(valid_graph_index, compact=True)
        context_graph = context_masking_graph.graph_mask(valid_graph_index, compact=True)
        context_substruct_overlap_graph = graph.subgraph(context_substruct_overlap_masking).graph_mask(
            masking_graph_index, compact=True)

        return substruct_graph, context_graph, context_substruct_overlap_graph

    def predict_and_target(self, batch):
        substruct_graph, context_graph, context_substruct_overlap_graph = self.extract_substruct_context_graph(batch)

        substruct_output = self.model(substruct_graph, substruct_graph.node_feature.long(),
                                      substruct_graph.edge_feature.long())
        center_node_idx = substruct_graph.center_node_masking.nonzero()[:, 0]
        substruct_repr = substruct_output['node_feature'][center_node_idx]

        overlap_output = self.model_context(context_graph, context_graph.node_feature.long(),
                                            context_graph.edge_feature.long())
        context_substruct_overlap_masking = context_graph.context_substruct_overlap_masking.nonzero()[:, 0]
        overlap_repr = overlap_output['node_feature'][context_substruct_overlap_masking]

        context_repr = self.model_context_readout(context_substruct_overlap_graph, overlap_repr)

        neg_context_repr = \
            torch.cat([context_repr[self.cycle_index(len(context_repr), i + 1)] for i in range(self.neg_sample_ratio)], dim=0)

        pred_pos = torch.sum(substruct_repr * context_repr, dim=1)
        pred_neg = torch.sum(substruct_repr.repeat((self.neg_sample_ratio, 1)) * neg_context_repr, dim=1)

        pred = torch.cat([pred_pos, pred_neg], dim=0)
        target = torch.cat([torch.ones_like(pred_pos), torch.zeros_like(pred_neg)], dim=0)
        return pred, target

    def evaluate(self, pred, target):
        metric = {}
        acc = torch.mean(((pred > 0).float() == (target == 1).float()).float())
        metric['acc'] = acc
        return metric
